(defn lb-min-conn
  ([pool] (lb-min-conn :lb-minn-conn pool))
  ([name pool] (lb-min-conn name {} pool))
  ([name opts pool]
   (let [error-hold-time (get opts :error-hold-time 0)
         conns (atom (apply sorted-set
                            (map (fn [idx] [0 idx])
                                 (range (count pool)))))
         ; Grab a connection.
         acquire (fn acquire []
                   (let [a (atom nil)]
                     (swap! conns
                            (fn acquire-swap [conns]
                              (let [[count idx :as conn] (first conns)
                                    conns (-> conns
                                            (disj conn)
                                            (conj [(inc count) idx]))] 
                                (reset! a idx)
                                conns)))
                     @a))
